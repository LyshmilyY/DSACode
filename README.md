```shell

```

``` shell
# 切换工作路径
cd LinearList
# 编译
make
# 运行 SqList 顺序表示例
make run_SqList 
# 运行 SlList 单链表示例
make run_SlList
# 运行 DlList 双链表示例
make run_DlList
# 运行 LinearList 全部实例
make run
# 清理文件
make clean
```
